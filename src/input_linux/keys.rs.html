impl Key {
    pub fn is_button(&self) -> bool {
        let key = *self as u16;
        ((key >= Key::Button0 as _) && (key < Key::KeyOk as _)) || key >= Key::ButtonTriggerHappy1 as _
    }

    pub fn is_key(&self) -> bool {
        !self.is_button()
    }
}
